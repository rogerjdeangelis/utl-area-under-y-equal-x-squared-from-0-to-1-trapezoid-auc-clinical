    %utl_submit_r64x('
    library(pracma);
    df<-data.frame(X=double(),Y=double());
    for (i in seq(0,20,1)) {
       df[i,"X"]=i/20;
       df[i,"Y"]=(i/20)^2;
       };
    df;
    AUC = trapz(df$X,df$Y);
    cat("Area under x squared from 0 to 1 = ",AUC);
    ');
